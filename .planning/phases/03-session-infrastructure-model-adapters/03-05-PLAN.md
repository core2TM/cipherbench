        runner = create_model_session(seed=SEED, difficulty=EASY, adapter=adapter, output_dir=tmp_path)
        session = runner.run()
        if reference_outcome is None: reference_outcome = session["outcome"]
        assert session["outcome"] == reference_outcome, f"Run {run}: got {session['outcome']}, expected {reference_outcome}. State bleed detected."
