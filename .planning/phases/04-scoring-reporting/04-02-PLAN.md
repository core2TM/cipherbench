    load_sessions(sessions_dir: Path, runner_type: str, model: str | None = None, difficulty: str | None = None) -> list[dict]:
    - Guard: if not sessions_dir.exists(): return [] (Pitfall 4)
    - Loop over sessions_dir.glob("*.json")
    - try/except (json.JSONDecodeError, OSError): continue on each file (Pitfall 2, T-04-03 malformed JSON guard)
    - Filter: data.get("outcome") not in TERMINAL_OUTCOMES → continue (D-04, Pitfall 2)
    - Filter: data.get("runner_type") != runner_type → continue (D-07 Phase 3)
    - Filter: model is not None and data.get("model") != model → continue (Pitfall 3: exact-match, no slug)
    - Filter: difficulty is not None and data.get("difficulty") != difficulty → continue (D-05)
    - Append and return

    efficiency_score(session: dict) -> float:
    - success = 1 if session["outcome"] == "success" else 0
    - attempts_used = sum(1 for a in session["attempts"] if not a.get("extraction_failed", False)) (D-06)
    - return success * (MAX_ATTEMPTS - attempts_used + 1) / MAX_ATTEMPTS (SCORE-02 formula)

    success_rate(sessions: list[dict]) -> float:
    - if not sessions: return 0.0 (SCORE-01)
    - return sum(1 for s in sessions if s["outcome"] == "success") / len(sessions)

    group_by_difficulty(sessions: list[dict]) -> dict[str, list[dict]]:
    - groups: dict[str, list[dict]] = {}
    - for s in sessions: tier = s.get("difficulty", "unknown"); groups.setdefault(tier, []).append(s)
    - return groups (SCORE-04)

    agi_proximity(model_sessions: list[dict], human_sessions: list[dict]) -> float | None:
    - if not human_sessions: return None (D-10)
    - model_avg = mean of efficiency_score(s) for s in model_sessions; if not model_sessions: model_avg = 0.0
    - human_avg = mean of efficiency_score(s) for s in human_sessions
    - if human_avg == 0.0: return None (Pitfall 5: avoid ZeroDivisionError)
    - return model_avg / human_avg (D-09, SCORE-03)

    compute_report(model_sessions: list[dict], human_sessions: list[dict], model_str: str | None = None) -> ScoreReport:
    - group model sessions by difficulty tier via group_by_difficulty
    - group human sessions by difficulty tier via group_by_difficulty
    - for each tier in model groups: compute sessions count, success_rate, avg_efficiency, agi_proximity (using that tier's human group)
    - compute totals row across all model sessions and all human sessions
    - generated_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
    - return ScoreReport dict with model=model_str, sessions_scored=len(model_sessions), by_difficulty=..., totals=..., generated_at=...
